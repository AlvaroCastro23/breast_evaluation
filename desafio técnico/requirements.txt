flask              # Web framework simples para criar a API e o front-end de upload

# === Deep Learning ===

# === Visualização (apenas para treino/relatório) ===
matplotlib         # Geração de gráficos de treino e métricas

# === Deploy (opcional, produção) ===
gunicorn           # Servidor WSGI recomendado para correr Flask em produção
